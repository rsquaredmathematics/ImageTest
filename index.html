\( \begin{aligned} & \mathrm{f}\left(\mathrm{x}_{1}\right)=\mathrm{f}\left(\mathrm{x}_{2}\right) \\ & \Rightarrow \frac{x_{1}}{x_{1}^{2}+1}=\frac{x_{2}}{x_{2}^{2}+1} \\ & \Rightarrow \mathrm{x}_{1} \mathrm{x}_{2}\left(\mathrm{x}_{2}-\mathrm{x}_{1}\right)=\mathrm{x}_{2}-\mathrm{x}_{1} \\ & \Rightarrow \mathrm{x}_{2} \mathrm{X}_{1}=1 \text {, if } \mathrm{x}_{2} \neq \mathrm{x}_{1} \end{aligned} \).
